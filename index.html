// ========================================== // ⚙️ 設定：ここを変えるとゲームのルールが変わるよ！ // ========================================== const TARGET_COUNT = 100; // 目標のステップ回数 // ========================================== let stepCount = 0; let lastStepTime = 0; let currentDirection = "center"; const ANGLE_THRESHOLD = 15; // 傾きの判定角度（度） document.getElementById('left-count').textContent = TARGET_COUNT; function startApp() { // iPhone用の許可申請（Androidではスキップされて下のelseに進みます） if (typeof DeviceOrientationEvent !== 'undefined' && typeof DeviceOrientationEvent.requestPermission === 'function') { DeviceOrientationEvent.requestPermission().then(permissionState => { if (permissionState === 'granted') { initSensor(); } else { alert("センサーのきょかが ないと うごかないよ！"); } }).catch(console.error); } else { // Pixel(Android)は直接ここが実行されます initSensor(); } } function initSensor() { window.addEventListener('deviceorientation', handleOrientation, true); document.getElementById('cheer-msg').textContent = "レッツ・ステップ！ふみだそう！"; } function handleOrientation(event) { // AndroidとiPhoneの両方で「左右の傾き」を正しく取得するための処理 let gamma = event.gamma; if (gamma === null || gamma === undefined) return; let now = Date.now(); // 左右15度以上の傾きを検知 if (gamma > ANGLE_THRESHOLD && currentDirection !== "right") { processStep(now); currentDirection = "right"; } else if (gamma < -ANGLE_THRESHOLD && currentDirection !== "left") { processStep(now); currentDirection = "left"; } else if (Math.abs(gamma) < 5) { currentDirection = "center"; } } function processStep(now) { if (lastStepTime !== 0 && (now - lastStepTime) < 600) { showWarning(); lastStepTime = now; return; } hideWarning(); stepCount++; lastStepTime = now; document.getElementById('count-box').textContent = stepCount; let left = TARGET_COUNT - stepCount; document.getElementById('left-count').textContent = left >= 0 ? left : 0; updateCheerMessage(); if (stepCount >= TARGET_COUNT) { gameClear(); } } function updateCheerMessage() { let progress = stepCount / TARGET_COUNT; if (progress >= 0.75) { document.getElementById('cheer-msg').innerHTML = "🔥 あとちょっと！ラストスパートだ！ 🔥"; } else if (progress >= 0.5) { document.getElementById('cheer-msg').innerHTML = "✨ はんぶん きたよ！そのいきだ！ ✨"; } else if (progress >= 0.25) { document.getElementById('cheer-msg').innerHTML = "👍 いいね！リズムに のってきた！ 👍"; } } function showWarning() { document.getElementById('warn-msg').style.visibility = "visible"; } function hideWarning() { document.getElementById('warn-msg').style.visibility = "hidden"; } function gameClear() { window.removeEventListener('deviceorientation', handleOrientation); document.getElementById('game-main').style.display = "none"; document.getElementById('game-clear').style.display = "block"; let randomPin = Math.floor(1000 + Math.random() * 9000); document.getElementById('password-display').textContent = "【 " + randomPin + " 】"; }
